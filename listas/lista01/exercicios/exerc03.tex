\item\label{norms}
Determine se a sequência dada é crescente, decrescente ou não
monótona. A sequência é limitada?

\begin{multicols}{3}
    \begin{enumerate}
        \item $a_n = (-2)^{n+1}$
        \item $a_n = \dfrac{1}{n^2 + 3}$
        \item $a_n = \dfrac{2n - 3}{3n + 4}$
        \item $a_n = n(-1)^n$
        \item $a_n = ne^{-n}$
        \item $a_n = \dfrac{n}{n^2 + 1}$
        \item $a_n = n + \dfrac{1}{n}$
    \end{enumerate}
\end{multicols}
